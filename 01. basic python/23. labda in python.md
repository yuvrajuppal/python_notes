```python
# Square numbers
numbers = [1, 2, 3, 4]
squared = list(map(lambda x: x**2, numbers))
print(squared)  # [1, 4, 9, 16]

# Double values
doubled = list(map(lambda x: x * 2, [5, 10, 15]))
print(doubled)  # [10, 20, 30]


# Add two lists element-wise
list1 = [1, 2, 3]
list2 = [4, 5, 6]
result = list(map(lambda x, y: x + y, list1, list2))
print(result)  # [5, 7, 9]

# With strings and numbers
names = ['alice', 'bob']
lengths = list(map(lambda name: len(name), names))
print(lengths)  # [5, 3]

```


```python
# Even numbers
numbers = [1, 2, 3, 4, 5, 6]
evens = list(filter(lambda x: x % 2 == 0, numbers))
print(evens)  # [2, 4, 6]

# Positive values
values = [-1, 0, 1, 2, -3, 4]
positives = list(filter(lambda x: x > 0, values))
print(positives)  # [1, 2, 4]

# Multiples of 3 and less than 10
nums = range(15)
multiples_3 = list(filter(lambda x: x % 3 == 0 and x < 10, nums))
print(multiples_3)  # [0, 3, 6, 9]

# Not empty strings
mixed = ['', 'hello', '', 'world', ' ']
non_empty = list(filter(lambda s: s.strip() != '', mixed))
print(non_empty)  # ['hello', 'world']

```


```python
# Basic single arg
square = lambda x: x ** 2
print(square(4))  # 16

# Multiple args
multiply = lambda x, y: x * y
print(multiply(3, 5))  # 15

# No args
greet = lambda: "Hello"
print(greet())  # Hello

```


```python
# Sort by length
fruits = ['apple', 'banana', 'kiwi']
print(sorted(fruits, key=lambda f: len(f)))  # ['kiwi', 'apple', 'banana']

# Sort dict by value
data = {'a': 30, 'b': 10, 'c': 20}
print(sorted(data.items(), key=lambda x: x[1]))  # [('b', 10), ('c', 20), ('a', 30)]

# Descending by second element
pairs = [(1, 5), (2, 3), (3, 8)]
print(sorted(pairs, key=lambda x: x[1], reverse=True))  # [(3, 8), (1, 5), (2, 3)]

```


```python
from functools import reduce

# Sum all numbers
nums = [1, 2, 3, 4]
total = reduce(lambda x, y: x + y, nums)
print(total)  # 10

# Find maximum
values = [5, 2, 8, 1, 9]
max_val = reduce(lambda x, y: x if x > y else y, values)
print(max_val)  # 9

# String concatenation
words = ['hello', ' ', 'world']
result = reduce(lambda x, y: x + y, words)
print(result)  # 'hello world'

```
