```python
"user": "root",
"password": "",

conn = mysql.connector.connect(**config)
```

example via user input
```python
from database import pool,conn

try:
	username = input("enter your name: ")
	user_email = input("enter your user email")
	user_age = int(input("enter your age"))
	sql = '''insert into studentdata (name,email,age) values (%s,%s,%s)'''
  
	pool.execute(sql,(username,user_email,user_age))
	conn.commit()
	print("done")
except Exception as err:
	print(f"error {err}")
```
